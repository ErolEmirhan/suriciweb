// Firebase Firestore Security Rules for Online Orders
// Bu kuralları Firebase Console > Firestore Database > Rules bölümüne yapıştırın
// Proje: makaraonline-5464e

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Online siparişler koleksiyonu - Herkes yazabilir, okuma sadece admin için
    match /orders/{orderId} {
      // Herkes sipariş oluşturabilir (yazma)
      allow create: if true;
      
      // Okuma için authentication gerekli (admin paneli için)
      allow read: if request.auth != null;
      
      // Güncelleme ve silme sadece authenticated kullanıcılar için
      allow update, delete: if request.auth != null;
    }
    
    // Diğer koleksiyonlar için varsayılan kurallar
    match /{document=**} {
      allow read, write: if false;
    }
  }
}
